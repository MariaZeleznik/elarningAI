## Krok 1: Czyszczenie danych

```python
def clean_retail_data(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()
    
    # Usuń zwroty (ujemna ilość) i anulowane faktury
    df = df[~df['InvoiceNo'].astype(str).str.startswith('C')]
    df = df[df['Quantity'] > 0]
    df = df[df['UnitPrice'] > 0]
    
    # Usuń wiersze bez CustomerID
    df = df.dropna(subset=['CustomerID'])
    df['CustomerID'] = df['CustomerID'].astype(int)
    
    # Kolumna: wartość wiersza
    df['Revenue'] = df['Quantity'] * df['UnitPrice']
    
    # Parsuj datę
    df['InvoiceDate'] = pd.to_datetime(df['InvoiceDate'])
    df['Year'] = df['InvoiceDate'].dt.year
    df['Month'] = df['InvoiceDate'].dt.month
    df['DayOfWeek'] = df['InvoiceDate'].dt.day_name()
    df['Hour'] = df['InvoiceDate'].dt.hour
    
    print(f"Po czyszczeniu: {len(df):,} wierszy (było {len(df):,})")
    return df

df_clean = clean_retail_data(df)
```

## Krok 3: Analiza RFM - segmentacja klientów

```python
ref_date = df_clean['InvoiceDate'].max() + timedelta(days=1)

rfm = df_clean.groupby('CustomerID').agg(
    Recency=('InvoiceDate', lambda x: (ref_date - x.max()).days),
    Frequency=('InvoiceNo', 'nunique'),
    Monetary=('Revenue', 'sum'),
).reset_index()

# Kwantylowe segmentowanie (1=najgorszy, 5=najlepszy)
def rfm_segment(df):
    df['R_Score'] = pd.qcut(df['Recency'], q=5, labels=[5,4,3,2,1]).astype(int)
    df['F_Score'] = pd.qcut(df['Frequency'].rank(method='first'), q=5, labels=[1,2,3,4,5]).astype(int)
    df['M_Score'] = pd.qcut(df['Monetary'].rank(method='first'), q=5, labels=[1,2,3,4,5]).astype(int)
    df['RFM_Score'] = df['R_Score'] * 100 + df['F_Score'] * 10 + df['M_Score']
    return df

rfm = rfm_segment(rfm)

# Segmentacja klientów
def assign_segment(row):
    r, f = row['R_Score'], row['F_Score']
    if r >= 4 and f >= 4: return 'Champions'
    elif r >= 3 and f >= 3: return 'Loyal Customers'
    elif r >= 4 and f < 2: return 'New Customers'
    elif r < 2 and f >= 3: return 'At Risk'
    elif r < 2 and f < 2: return 'Lost'
    else: return 'Potential Loyalists'

rfm['Segment'] = rfm.apply(assign_segment, axis=1)

print("\n=== SEGMENTACJA KLIENTÓW ===")
seg_stats = rfm.groupby('Segment').agg(
    Klientów=('CustomerID', 'count'),
    Udział=('CustomerID', lambda x: f"{len(x)/len(rfm):.1%}"),
    ŚrPrzychód=('Monetary', lambda x: f"{x.mean():,.0f} GBP"),
).sort_values('Klientów', ascending=False)
print(seg_stats)
```

## Krok 4: ABC Analysis produktów

```python
product_revenue = df_clean.groupby('Description')['Revenue'].sum().sort_values(ascending=False)
product_revenue_cum = product_revenue.cumsum() / product_revenue.sum()

def abc_category(cum_pct):
    if cum_pct <= 0.70: return 'A'  # 70% przychodu
    elif cum_pct <= 0.90: return 'B'  # kolejne 20%
    else: return 'C'  # ostatnie 10%

abc = pd.DataFrame({
    'Produkt': product_revenue.index,
    'Przychód': product_revenue.values,
    'Udział skumulowany': product_revenue_cum.values,
})
abc['Kategoria ABC'] = abc['Udział skumulowany'].apply(abc_category)

print("\n=== ANALIZA ABC PRODUKTÓW ===")
print(abc.groupby('Kategoria ABC').agg(
    Produktów=('Produkt', 'count'),
    Przychód=('Przychód', 'sum'),
).assign(Udział=lambda x: (x['Przychód'] / x['Przychód'].sum()).map('{:.1%}'.format)))
```

## Krok 5: Prognoza sprzedaży

```python
from sklearn.linear_model import LinearRegression
import numpy as np

monthly = df_clean.groupby(['Year', 'Month'])['Revenue'].sum().reset_index()
monthly['Period'] = range(len(monthly))

X = monthly[['Period']].values
y = monthly['Revenue'].values

model = LinearRegression()
model.fit(X, y)

# Prognoza: 3 kolejne miesiące
future_periods = np.array([[i] for i in range(len(monthly), len(monthly)+3)])
forecast = model.predict(future_periods)

print("\n=== PROGNOZA NA 3 MIESIĄCE ===")
for i, (period, pred) in enumerate(zip(future_periods, forecast)):
    print(f"Miesiąc +{i+1}: {pred:,.0f} GBP")

r2 = model.score(X, y)
print(f"\nR² modelu trendu: {r2:.3f}")
```
