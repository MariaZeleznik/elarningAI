from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# Przygotuj features dla każdego filmu
movies['genre_str'] = movies['genres'].str.replace('|', ' ')

tfidf = TfidfVectorizer(stop_words='english')
tfidf_matrix = tfidf.fit_transform(movies['genre_str'])

# Macierz podobieństwa (cosine similarity)
cosine_sim = cosine_similarity(tfidf_matrix, tfidf_matrix)

def get_similar_movies(title, n=10):
    if title not in movies['title'].values:
        return f"Nie znaleziono: {title}"
    
    idx = movies[movies['title'] == title].index[0]
    scores = list(enumerate(cosine_sim[idx]))
    scores = sorted(scores, key=lambda x: x[1], reverse=True)[1:n+1]
    
    movie_indices = [i[0] for i in scores]
    result = movies.iloc[movie_indices][['title', 'genres']].copy()
    result['similarity'] = [s[1] for s in scores]
    return result

print(get_similar_movies("Toy Story (1995)", n=5))

## Krok 3: Collaborative Filtering z Matrix Factorization

```python
import torch
import torch.nn as nn
from torch.utils.data import Dataset, DataLoader

class RatingsDataset(Dataset):
    def __init__(self, ratings_df):
        self.users = torch.LongTensor(ratings_df['user_id'].values - 1)
        self.movies = torch.LongTensor(ratings_df['movie_id'].values - 1)
        # Normalizuj: 1-5 → 0-1
        self.ratings = torch.FloatTensor((ratings_df['rating'].values - 1) / 4)
    
    def __len__(self): return len(self.ratings)
    def __getitem__(self, idx): return self.users[idx], self.movies[idx], self.ratings[idx]

class MatrixFactorization(nn.Module):
    def __init__(self, n_users, n_movies, n_factors=50):
        super().__init__()
        self.user_emb = nn.Embedding(n_users, n_factors)
        self.movie_emb = nn.Embedding(n_movies, n_factors)
        self.user_bias = nn.Embedding(n_users, 1)
        self.movie_bias = nn.Embedding(n_movies, 1)
        
        nn.init.normal_(self.user_emb.weight, std=0.01)
        nn.init.normal_(self.movie_emb.weight, std=0.01)
        nn.init.zeros_(self.user_bias.weight)
        nn.init.zeros_(self.movie_bias.weight)
    
    def forward(self, user_ids, movie_ids):
        u = self.user_emb(user_ids)
        m = self.movie_emb(movie_ids)
        ub = self.user_bias(user_ids).squeeze()
        mb = self.movie_bias(movie_ids).squeeze()
        return torch.sigmoid((u * m).sum(dim=1) + ub + mb)

n_users = ratings['user_id'].max()
n_movies = ratings['movie_id'].max()

model = MatrixFactorization(n_users, n_movies, n_factors=50)
dataset = RatingsDataset(ratings.sample(100000, random_state=42))  # subset dla szybkości
loader = DataLoader(dataset, batch_size=2048, shuffle=True)

optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
criterion = nn.MSELoss()

for epoch in range(5):
    model.train()
    total_loss = 0
    for user_ids, movie_ids, ratings_batch in loader:
        optimizer.zero_grad()
        pred = model(user_ids, movie_ids)
        loss = criterion(pred, ratings_batch)
        loss.backward()
        optimizer.step()
        total_loss += loss.item()
    
    print(f"Epoch {epoch+1}: avg loss = {total_loss/len(loader):.4f}")
```

## Metryki ewaluacji systemów rekomendacji

```python
def precision_at_k(recommended, relevant, k=10):
    recommended_k = recommended[:k]
    hits = len(set(recommended_k) & set(relevant))
    return hits / k

def recall_at_k(recommended, relevant, k=10):
    recommended_k = recommended[:k]
    hits = len(set(recommended_k) & set(relevant))
    return hits / len(relevant) if relevant else 0

def ndcg_at_k(recommended, relevant, k=10):
    dcg = sum([1/np.log2(i+2) for i, movie in enumerate(recommended[:k]) if movie in relevant])
    ideal_dcg = sum([1/np.log2(i+2) for i in range(min(k, len(relevant)))])
    return dcg / ideal_dcg if ideal_dcg > 0 else 0
```
